# أوامر تحسين بناء Flutter للسرعة القصوى

## 1. تنظيف المشروع أولاً
flutter clean
flutter pub get

## 2. بناء APK محسن للسرعة (Debug)
flutter build apk --debug --target-platform android-arm64 --no-tree-shake-icons

## 3. بناء APK للجهاز المحدد فقط (أسرع)
flutter build apk --debug --target-platform android-arm64 --split-per-abi

## 4. تشغيل مباشر على الجهاز (الأسرع)
flutter run --debug --hot --target-platform android-arm64

## 5. بناء Bundle بدلاً من APK (أصغر حجماً)
flutter build appbundle --debug

## 6. تحسينات إضافية للتطوير
flutter run --debug --enable-software-rendering
flutter run --debug --disable-service-auth-codes

## 7. فحص حجم التطبيق
flutter build apk --debug --analyze-size

## ملاحظات مهمة:
# - استخدم --target-platform android-arm64 للأجهزة الحديثة
# - استخدم --split-per-abi لتقليل حجم APK
# - استخدم flutter run بدلاً من build apk للتطوير السريع
# - تأكد من تفعيل USB Debugging على الجهاز
# - استخدم كابل USB جيد وسريع
